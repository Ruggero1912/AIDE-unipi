from sklearn.datasets import make_blobs
X, y = make_blobs(n_samples=1000, 
                  centers=5,
                  cluster_std = [0.1,0.1,0.1,0.1,1], 
                  center_box = (-10,10), 
                  n_features=2, 
                  random_state=23)
plt.figure(figsize=(10,10))
plt.scatter([x[0] for x in X],[x[1] for x in X],c = y)
plt.xlim([-10,10])
plt.ylim([-10,10])
